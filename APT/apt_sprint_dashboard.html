📐 UEBA Gaussian Anomaly Scoring

@Service public class UEBAProfiler { // μ = (1/N) Σ xᵢ | σ² = (1/N) Σ (xᵢ - μ)² public double computeAnomalyScore(String entityId, String feature, double value) { Baseline b = baselineRepo.findByEntityAndFeature(entityId, feature); double mu = b.getMu(); double sigma = Math.sqrt(b.getSigmaSq()); // Three-sigma rule: flag if |x - μ| > 3σ double deviation = Math.abs(value - mu) / (sigma + 1e-9); if (deviation > 3.0) { // Gaussian probability P(x) using normal distribution double px = gaussianPDF(value, mu, sigma); return px < EPSILON ? 1.0 : deviation / 10.0; } return 0.0; } // Update rolling baseline with Welford's online algorithm public void updateBaseline(String entityId, String feature, double newValue) { Baseline b = baselineRepo.findOrCreate(entityId, feature); long n = b.getSampleCount() + 1; double delta = newValue - b.getMu(); double newMu = b.getMu() + delta / n; double delta2 = newValue - newMu; b.update(n, newMu, b.getSigmaSq() + (delta * delta2 - b.getSigmaSq()) / n); baselineRepo.save(b); } }

🌲 Isolation Forest + Autoencoder (DL4J)

@Service public class AnomalyDetectionEngine { private MultiLayerNetwork autoencoder; // DL4J model public DetectionResult detect(INDArray features) { // 1. Isolation Forest: short path = anomaly double ifScore = isolationForest.scoreInstance(features); // 2. Autoencoder: high reconstruction error = anomaly INDArray reconstructed = autoencoder.output(features); double reconError = features.squaredDistance(reconstructed); // 3. Peer-group deviation from UEBA baselines double peerDeviation = uebaProfiler.peerGroupDeviation(features); // Unified risk score 0-100 double riskScore = (0.4 * normalize(ifScore) + 0.4 * normalize(reconError) + 0.2 * normalize(peerDeviation)) * 100; return new DetectionResult(riskScore, ifScore, reconError); } private MultiLayerNetwork buildAutoencoder() { var conf = new NeuralNetConfiguration.Builder() .updater(new Adam(0.001)) .list() .layer(new DenseLayer.Builder().nIn(20).nOut(12).activation(Activation.RELU).build()) .layer(new DenseLayer.Builder().nIn(12).nOut(6).activation(Activation.RELU).build()) // bottleneck .layer(new DenseLayer.Builder().nIn(6).nOut(12).activation(Activation.RELU).build()) .layer(new OutputLayer.Builder().nIn(12).nOut(20).lossFunction(LossFunctions.MSE).build()) .build(); return new MultiLayerNetwork(conf); } }

🗺️ MITRE ATT&CK Mapping Logic

@Component public class MitreMapper { // Rules engine: feature pattern → ATT&CK technique private static final Map<String, String> RULES = Map.of( "failed_login_spike", "T1110 - Brute Force", "lateral_smb_traffic", "T1021 - Remote Services", "unusual_login_hour", "T1078 - Valid Accounts", "large_data_upload", "T1041 - Exfiltration over C2", "dns_tunneling", "T1071 - Application Layer Protocol", "new_admin_account", "T1136 - Create Account" ); public MitreTechnique map(NetworkEvent event, DetectionResult result) { return RULES.entrySet().stream() .filter(e -> matchesPattern(event, result, e.getKey())) .findFirst() .map(e -> new MitreTechnique(e.getValue())) .orElse(MitreTechnique.UNKNOWN); } }
